library IEEE;
use IEEE.STD_LOGIC_1164.ALL;

entity DEMUX_1to4 is
    Port (
        D  : in  STD_LOGIC;
        S  : in  STD_LOGIC_VECTOR(1 downto 0);
        Y0 : out STD_LOGIC;
        Y1 : out STD_LOGIC;
        Y2 : out STD_LOGIC;
        Y3 : out STD_LOGIC
    );
end DEMUX_1to4;

architecture Behavioral of DEMUX_1to4 is
begin
    Y0 <= D when (S="00") else '0';
    Y1 <= D when (S="01") else '0';
    Y2 <= D when (S="10") else '0';
    Y3 <= D when (S="11") else '0';
end Behavioral;
